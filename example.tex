\documentclass{article}

\usepackage{bayesdiagram}
\usetikzlibrary{matrix}

\begin{document}

\begin{tikzpicture}[node distance=3cm and 2cm, >=stealth]

\randomvar[gaussian]                               (M0) {M_0};
\randomvar[exponential, node/.style={right of=M0}] (M1) {M_1};

\bayesequation[below of=M1] (eqn) {\param{\beta_0} + \param{\beta_1} x_i};

\randomvar[gamma, node/.style={right of=eqn}] (M2) {M_2};
\randomvar[normal, node/.style={right of=M2}] (M4) {M_4};  
\randomvar[normal, node/.style={below of=eqn}] (mui) {\mu_i};

\draw[->] (eqn) -- node [anchor=east] {$=$} (mui.center);          
\draw[->] (M0) -- node [anchor=east] {$\sim$} (eqn parameter 1);
\draw[->] (M1) -- node [anchor=east] {$\sim$} (eqn parameter 2);
\draw[->] (M2) -- node [anchor=east] {$\sim$} (mui parameter 1);
\draw[->] (M4) -- node [anchor=east] {$\sim$} (mui parameter 2);

\end{tikzpicture}

\end{document}
