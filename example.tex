\documentclass{article}

\usepackage{bayesdiagram}
\usetikzlibrary{matrix}

\begin{document}

\begin{tikzpicture}[node distance=3cm and 2cm, >=stealth]

\randomvar[gaussian]                               (H0) {M_0, T_0};
\randomvar[exponential, node/.style={right of=H0}] (H1) {M_1, T_1};

\bayesequation[below of=H1] (eqn) {\param{\beta_0} + \param{\beta_1} x_i};

\randomvar[gamma, node/.style={right of=eqn}] (H2) {S, R};
\randomvar[normal, node/.style={right of=H2}] (H4) {\nu, \sigma};  
\randomvar[normal, node/.style={below of=eqn}] (H5) {\mu_i, \tau};

\draw[->] (eqn) -- node [anchor=east] {$=$} (H5.center);          
\draw[->] (H0) -- node [anchor=east] {$\sim$} (eqn parameter 1);
\draw[->] (H1) -- node [anchor=east] {$\sim$} (eqn parameter 2);
\draw[->] (H2) -- node [anchor=east] {$\sim$} (H5 parameter 1);
\draw[->] (H4) -- node [anchor=east] {$\sim$} (H5 parameter 2);

\end{tikzpicture}

\end{document}
