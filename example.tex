\documentclass{article}

\usepackage{bayesdiagram}
\usepackage{sfmath}

\begin{document}

\begin{tikzpicture}[node distance=3cm and 2cm, >=stealth]

\distribution[gaussian]                 (H1) {M_0, T_0};
\distribution[exponential, right of=H1] (H2) {\lambda};

\bayesequation[below of=H2] (eqn) 
   {\param{\beta_0} + \param{\beta_1} \mathbf{x}_i};

\distribution[gamma, right of=eqn]      (H3) {S, R};
\distribution[normal, below of=eqn]     (H4) {\mu_i, \tau};

\draw[->] (eqn) -- node [anchor=east] {$=$} (H4 parameter 1);          
\draw[->] (H1) -- node [anchor=east] {$\sim$} (eqn parameter 1);
\draw[->] (H2) -- node [anchor=east] {$\sim$} (eqn parameter 2);
\draw[->] (H3) -- node [anchor=east] {$\sim$} (H4 parameter 2);

\end{tikzpicture}

\end{document}
